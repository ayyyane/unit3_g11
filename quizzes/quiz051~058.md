```.py
class palNum:
    def __init__(self, A:int, B:int):
        self.A = A
        self.B = B

    def get_palg_list(self):
        number = []
        for n in range(self.A, self.B+1):
            N = str(n)
            R = ""
            for i in N:
                R = i+R
            if R == N:
                number.append(int(N))

        return number

Number1 = palNum(1,100)
print(Number1.get_palg_list())
```

```.py
class rainDrops:

        ans = ""

        if self.n%3 == 0:

                ans += "Plang"
                if self.n%7 == 0:
                    ans += "Plong"
                    output += f"{self.n} is divisble by 3, 5, 7, so the result is {ans}"

                    output += f"{self.n} is divisible by 3,5, but not 7 so the result is {ans}"
```

```.py
class darts:
    def __init__(self, x:float, y:float):

        self.x = x
        self.y = y

    def calculate(self):
        R = (self.x**2) + (self.y**2)
        points = 0
        if R <= 1:
            points += 10
        elif 1<R and R<=25:
            points += 5
        elif 25<R and R<=100:
            points += 1

        return points


darts1 = darts(-5,9)
print(darts1.calculate())
```

```.py
def secret_handshake(code):

    actions = ["wink", "double blink", "close your eyes", "jump"]
    handshake = []

    for i in actions:
        if code % 2 == 1:
            handshake.append(i)
        code = code // 2

    if code > 0:
        handshake.reverse()

    return handshake

print(secret_handshake(9))
```
